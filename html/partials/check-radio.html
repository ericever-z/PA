checkbox/radio 用户自定义

默认样式
		
[type="checkbox"]{
	display: none;
}
[type="checkbox"]+label:before{
	position: absolute;
	display: inline-block;
	content:"";
	width: 16px;
    height: 16px;
	padding-left: 0.5px;
	line-height: 12px;
 	border:1px solid #b5b5b5;
	border-radius: 3px;
	bottom: 0;
	margin: 0 -20px;
	font-family: 'Glyphicons Halflings';
}
[type="checkbox"]+label:hover:before{
	box-shadow: 0 0 2px #c1c1c1;
}

[type="checkbox"]:checked+label:before{
	content:"\e013";/**** 对号 *****/
	box-shadow: 0 0 0;
	color:#7b7b7b;/**选中时的颜色***/
}

[type="checkbox"]+label{
	position: relative;
	left:20px;
}
/**** radio ****/
[type="radio"]{
	display: none;
}
[type="radio"]+label{
	margin: 0 10px;
}

[type="radio"]+label:before{
	position: absolute;
	display: inline-block;
	content:"";
	font-family: 'Glyphicons Halflings';	
	content:"";
	width: 14px;
    height: 14px;
	border-radius: 50%;
	line-height: 8px;
	border:1px solid #ccc;
	margin: 0 -18px;
	margin-top:1px;
	text-align: center;
}
[type="radio"]+label:hover:before{
	box-shadow: 0 0 2px #c1c1c1;
}
[type="radio"]:checked+label:before{
	content:"\2022"; /*** 选中时 里面的原点 ****/
	font-size: 15px;
	box-shadow: 0 0 0;
	color:#7b7b7b;/**选中时的颜色***/
}
