搜索输入框

普通搜索输入框 

		
<div class="ndp-search-wrapper" name="plain"></div>		
		
	
		
$(".ndp-search-wrapper[name=plain]").search({
	placeholder:"请输入字符"
});

搜索输入框--不可用 

		
<div class="ndp-search-wrapper" name="disable"></div>		
		
	
		
$(".ndp-search-wrapper[name=disable]").search({
	placeholder:"请输入字符",
	disabled:true
});

搜索输入框--带范围选择 

		
<div class="ndp-search-wrapper" name="mix"></div>		
		
	
		
$(".ndp-search-wrapper[name=mix]").search({
	type:2,
	dropList:[{text:"中国"},{text:"美国"},{text:"德国"}],
	disabled:true
}).on("search_input_change",function(e){

});
