面包屑控件bread.js

普通面包屑 

		
<div class="ndp-bread-wrapper" name="plain-bread"></div>
		
	
		
$("div.ndp-bread-wrapper[name=plain-bread]").bread({
	list:["Home","Library","Data"]
});

面包屑--自定义分隔符 

		
<div class="ndp-bread-wrapper" name="split-bread"></div>
		
	
		
$("div.ndp-bread-wrapper[name=split-bread]").bread({
	list:["Home","Library","Data"],
	spliter:">"
});

面包屑--自定义首页图标 

		
<div class="ndp-bread-wrapper" name="home-bread"></div>
		
	
		
$("div.ndp-bread-wrapper[name=home-bread]").bread({
	home:"<i class='glyphicon glyphicon-home'></i>",
	list:["Home","Library","Data"],
	spliter:">"
});
