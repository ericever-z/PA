列表插件table.js

普通列表 

		
<div class="ndp-table-wrapper" name="plain-table"></div>
		
	
		
$(".ndp-table-wrapper[name='plain-table']").table({
	head:{col1:"第一列",col2:"第二列",col3:"第三列",col4:"第四列"},//列表头数据，
	//列表头的数据 支持 数组里面是字符串
	//列如：head:{"第一列","第二列","第三列","第四列"}
	//也支持 混排，字符串和{} 都有的形式， 下面的列表数据也是一样
	data:[//列表数据，数据格式数组，数组里面可以是{}  也可以是字符串或者数字
		{id:1101,name:"hello123",info:"hello -china 123",gender:"man"},
		{id:1102,name:"hello_23",info:"hello - japan 34",gender:"woman"},
		{id:1103,name:"hello_34",info:"hello - euro 23",gender:"no sign"}
	]	
});

列表--固定选中行 

	
<div class="ndp-table-wrapper" name="solid-row-table"></div>
	
	
	
$(".ndp-table-wrapper[name='solid-row-table']").table({
	head:["第一列","第二列","第三列","第四列"],
	data:[{id:1101,name:"hello123",info:"hello -china 123",gender:"man"},
		  {id:1102,name:"hello_23",info:"hello - japan 34",gender:"woman"},
		  {id:1103,name:"hello_34",info:"hello - euro 23",gender:"no sign"}
		  ],
	rowNail:true,//允许选中行
	activeRow:1//默认选中的行索引，base 0
});
